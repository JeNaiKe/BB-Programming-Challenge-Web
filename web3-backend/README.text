Create a file called "coinmarketcap.config.js" on the 'src/config' folder, and add the following code with your api key

const config = {
    CMC_API_URL: 'https://pro-api.coinmarketcap.com',
    CMC_API_KEY: "your-api-key-should-be-here",
    CMC_TEST_URL: "https://sandbox-api.coinmarketcap.com",
    CMC_TEST_KEY: "b54bcf4d-1bca-4e8e-9a24-22ff2c3d462c",
}

module.exports = config;
